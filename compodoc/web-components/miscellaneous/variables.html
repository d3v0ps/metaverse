findAncestor
Default value : (el: any, selectors: any) => { if (typeof el.closest === 'function') { return el.closest(selectors) || null; } while (el) { if (el.matches(selectors)) { return el; } el = el.parentElement; } return null; }

getEvent
Default value : ( event: MouseEvent | TouchEvent ): MouseEvent | Touch => { if (event.type === 'touchend' || event.type === 'touchcancel') { return (event as TouchEvent).changedTouches[0]; } return event.type.startsWith('touch') ? (event as TouchEvent).targetTouches[0] : (event as MouseEvent); }

isLTR
Default value : (): boolean => { let dir = 'ltr'; if (typeof window !== 'undefined') { if (window.getComputedStyle) { dir = window .getComputedStyle(document.body, null) .getPropertyValue('direction'); } else { // eslint-disable-next-line @typescript-eslint/no-explicit-any dir = (document.body as any).currentStyle.direction; } } return dir === 'ltr'; }

Returns whether the page is in LTR mode. Defaults to true if it can't be computed.
